\begin{frame}[t,shrink=8]
\NotesTitle{Notes \& Steps}
\footnotesize
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\KeyIdea{Rounding gives a simpler number that is close to the original. We round to a specific decimal place by looking at the next digit to the right.}
\SectionTitle{Steps}
\StepLine{1}{Find the decimal place you are rounding to.}
\StepLine{2}{Look at the digit immediately to the right.}
\StepLine{3}{If that digit is 5 or more, round up. If it is 4 or less, round down.}
\StepLine{4}{Drop all digits to the right of the rounding place.}
\end{column}
\begin{column}{0.48\textwidth}
\SectionTitle{Rounding reminder}
\begin{itemize}
  \setlength{\itemsep}{0.12em}
  \item Round \(3.4\) to nearest whole: look at \(4\) (tenths). \(4 < 5\), so \(3\).
  \item Round \(7.6\) to nearest whole: look at \(6\) (tenths). \(6 \ge 5\), so \(8\).
  \item Round \(2.31\) to 1 d.p.: look at \(1\) (hundredths). \(1 < 5\), so \(2.3\).
  \item Round \(5.68\) to 1 d.p.: look at \(8\) (hundredths). \(8 \ge 5\), so \(5.7\).
\end{itemize}
\end{column}
\end{columns}
\GreenDivider
\vfill
\begin{columns}[b,onlytextwidth]
\begin{column}{0.48\textwidth}
\CommonMistake{When the digit to round is a 9, rounding up can change the whole number. For example, \(2.97\) to 1 d.p. rounds to \(3.0\), not \(2.10\).}
\end{column}
\begin{column}{0.48\textwidth}
\TryThese{\begin{enumerate}
  \item Round \(4.86\) to 1 decimal place.
  \item Round \(1.73\) to 1 decimal place.
  \item Round \(6.15\) to 1 decimal place.
\end{enumerate}}
\end{column}
\end{columns}

\begin{frame}[t,shrink=12]
\NotesTitle{Notes \& Steps}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 1: round to 1 d.p.}{Round \(8.42\) to 1 decimal place.\[8.42\]The rounding digit is \(4\) (tenths). Look right at \(2\) (hundredths). \(2 < 5\), so the \(4\) stays. Answer: \(8.4\).}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 2: round to 2 d.p.}{Round \(3.141\) to 2 decimal places.\[3.141\]The rounding digit is \(4\) (hundredths). Look right at \(1\) (thousandths). \(1 < 5\), so the \(4\) stays. Answer: \(3.14\).}
\end{column}
\end{columns}
\GreenDivider
\TryThese{\begin{enumerate}
  \item Round \(5.678\) to 2 decimal places.
  \item Round \(7.204\) to 2 decimal places.
  \item Round \(9.995\) to 2 decimal places.
\end{enumerate}}
\end{frame}

\begin{frame}[t,shrink=12]
\NotesTitle{Notes \& Steps}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 3: 9 rolling over}{Round \(2.97\) to 1 decimal place.\[2.97\]The rounding digit is \(9\) (tenths). Look right at \(7\) (hundredths). \(7 \ge 5\), so we need to round the \(9\) up. \(9\) rounds to \(10\), which carries into the units: \(3.0\). Answer: \(3.0\).}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 4: 2 d.p. with 9}{Round \(4.995\) to 2 decimal places.\[4.995\]The rounding digit is \(9\) (hundredths). Look right at \(5\) (thousandths). \(5 \ge 5\), so round the \(9\) up. \(9\) becomes \(10\), carries: \(5.00\). Answer: \(5.00\).}
\end{column}
\end{columns}
\GreenDivider
\CommonMistake{Forgetting to show the zero after rounding. For example, \(3.0\) is correct for \(2.97\) to 1 d.p. If you write just \(3\), you have rounded to a whole number, not 1 decimal place.}
