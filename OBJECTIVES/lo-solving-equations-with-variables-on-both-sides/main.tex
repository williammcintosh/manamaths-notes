\begin{frame}[t,shrink=10]
\NotesTitle{Notes \& Steps}
\KeyIdea{Move variable terms to one side and constants to the other. Then solve.}
\vspace{-0.15em}
\footnotesize
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 1: basic}{Solve \(3x = 2x + 5\).\\
Step 1: Move \(2x\) (subtract from both sides)
\[3x - 2x = 2x + 5 - 2x\]
\[x = 5\]
Check: \(3\times5 = 2\times5+5 = 15\)}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 2: with brackets}{Solve \(5x - 3 = 2x + 9\).\\
Step 1: Move \(2x\) (subtract from both sides)
\[5x - 3 - 2x = 2x + 9 - 2x\]
\[3x - 3 = 9\]
Step 2: Add 3
\[3x = 12\]
Step 3: Divide by 3
\[x = 4\]
Check: \(5(4)-3 = 2(4)+9 = 17\)}
\end{column}
\end{columns}
\GreenDivider
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 3: variable on right}{Solve \(15 - 2a = 4a + 3\).\\
Step 1: Move \(-2a\) (add \(2a\) to both)
\[15 = 6a + 3\]
Step 2: Subtract 3
\[12 = 6a\]
Step 3: Divide by 6
\[2 = a\]
Check: \(15-2(2)=4(2)+3=11\)}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 4: negative variable}{Solve \(20 - 5n = 3n + 4\).\\
Step 1: Move \(-5n\) (add \(5n\))
\[20 = 8n + 4\]
Step 2: Subtract 4
\[16 = 8n\]
Step 3: Divide by 8
\[2 = n\]
Check: \(20-5(2)=3(2)+4=10\)}
\end{column}
\end{columns}
\GreenDivider
\begin{columns}[b,onlytextwidth]
\begin{column}{0.48\textwidth}
\TryThese{\begin{enumerate}
  \item \(3x = 2x + 5\)
  \item \(5a - 3 = 2a + 9\)
  \item \(15 - 2m = 4m + 3\)
  \item \(20 - 5n = 3n + 4\)
\end{enumerate}}
\end{column}
\begin{column}{0.48\textwidth}
\CommonMistake{Forgetting to move the sign. When moving \(2x\) from \(3x = 2x + 5\), subtract \(2x\) from both sides (not add). Keep track of positive/negative signs.}
\end{column}
\end{columns}
