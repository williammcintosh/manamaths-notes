\begin{frame}[t,shrink=10]
\NotesTitle{Notes \& Steps}
\footnotesize
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\KeyIdea{To expand double brackets, multiply every term in the first bracket by every term in the second bracket. Use FOIL: \textbf{F}irst, \textbf{O}uter, \textbf{I}nner, \textbf{L}ast. Then collect like terms.}
\SectionTitle{Steps}
\StepLine{1}{Multiply the First terms: \(x \times x = x^2\).}
\StepLine{2}{Multiply the Outer terms: \(x \times b = bx\).}
\StepLine{3}{Multiply the Inner terms: \(a \times x = ax\).}
\StepLine{4}{Multiply the Last terms: \(a \times b = ab\).}
\StepLine{5}{Add them up: \(x^2 + bx + ax + ab\), then simplify: \(x^2 + (a+b)x + ab\).}
\end{column}
\begin{column}{0.48\textwidth}
\SectionTitle{FOIL diagram}
\[
(x + a)(x + b)
\]
\begin{itemize}
  \item \textbf{F}: \(x \times x = x^2\)
  \item \textbf{O}: \(x \times b = bx\) (outer terms)
  \item \textbf{I}: \(a \times x = ax\) (inner terms)
  \item \textbf{L}: \(a \times b = ab\)
\end{itemize}
So \((x + a)(x + b) = x^2 + (a+b)x + ab\).
\end{column}
\end{columns}
\GreenDivider
\CommonMistake{Only doing First and Last (FO), skipping Outer and Inner. \((x+3)(x+4)=x^2+3x+4x+12=x^2+7x+12\), \textbf{not} \(x^2+12\).}
\end{frame}

\begin{frame}[t,shrink=12]
\NotesTitle{Notes \& Steps}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 1: both positive}{Expand \((x+2)(x+5)\). F: \(x^2\), O: \(5x\), I: \(2x\), L: \(10\). Sum: \(x^2+5x+2x+10 = x^2+7x+10\).}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 2: one negative}{Expand \((x+6)(x-3)\). F: \(x^2\), O: \(-3x\), I: \(6x\), L: \(-18\). Sum: \(x^2-3x+6x-18 = x^2+3x-18\).}
\end{column}
\end{columns}
\GreenDivider
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 3: both negative}{Expand \((x-4)(x-2)\). F: \(x^2\), O: \(-2x\), I: \(-4x\), L: \(8\). Sum: \(x^2-2x-4x+8 = x^2-6x+8\).}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 4: with coefficients}{Expand \((2x+1)(x+3)\). F: \(2x^2\), O: \(6x\), I: \(x\), L: \(3\). Sum: \(2x^2+6x+x+3 = 2x^2+7x+3\).}
\end{column}
\end{columns}
\GreenDivider
\begin{columns}[b,onlytextwidth]
\begin{column}{0.48\textwidth}
\TryThese{\begin{enumerate}
  \item \((x+3)(x+7)\)
  \item \((x-5)(x+2)\)
  \item \((2x+1)(x+4)\)
\end{enumerate}}
\end{column}
\begin{column}{0.48\textwidth}
\CommonMistake{Forgetting to collect like terms. Always combine the Outer and Inner terms (\(bx\) and \(ax\)) before writing your final answer.}
\end{column}
