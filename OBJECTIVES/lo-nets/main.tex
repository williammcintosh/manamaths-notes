\begin{frame}[t,shrink=12]
\NotesTitle{Notes -- Nets of 3D shapes}
\footnotesize
\KeyIdea{A net is a flat 2D plan that folds along edges to make a 3D shape. The same 3D shape can have many different nets. All faces must be present and connected.}
\vspace{-0.3em}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\SectionTitle{Steps to identify a net}
\StepLine{1}{Count the faces. Does the net have the right number? (Cube = 6, cuboid = 6, pyramid = 5, tetrahedron = 4)}
\StepLine{2}{Check shapes match. Are the faces the correct polygons?}
\StepLine{3}{Visualise folding. Imagine bending each face along the edge.}
\StepLine{4}{Check overlaps. No faces should overlap when folded.}
\StepLine{5}{Check gaps. There should be no gaps between faces.}
\end{column}
\begin{column}{0.48\textwidth}
\SectionTitle{Common nets}

A cube has 11 distinct nets. A cuboid also has a net of 6 rectangles (3 pairs of equal size).

\begin{frame}[t,shrink=6]
\NotesTitle{Notes -- Nets and surface area}
\vspace{-0.3em}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 1: Cube net}{A cube of side 3 cm has net area = $6 \times (3 \times 3) = 54$ cm$^2$. The net is the surface area of the cube.}
\vspace{0.3em}
\WorkedExample{Example 2: Cuboid box}{A 5 $\times$ 3 $\times$ 2 cm box: SA = $2(5\cdot3+5\cdot2+3\cdot2) = 62$ cm$^2$.}
\vspace{0.3em}
\WorkedExample{Example 3: Opposite faces}{In a cube cross net, the two squares at the ends of the cross are opposite each other when folded.}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 4: Open box}{An open box net has 5 faces (no top). Its surface area excludes the missing top face.}
\vspace{0.3em}
\TryThese{\begin{enumerate}
  \item Draw a cube net of side 2 cm.
  \item Find surface area of a 4 cm cube.
  \item Identify opposite faces in a cross net.
\end{enumerate}}
\end{column}
\end{columns}
\GreenDivider
\CommonMistake{Forgetting that opposite faces never share an edge in a net. In the cross net, the top and bottom squares are opposite.}
\end{frame}
