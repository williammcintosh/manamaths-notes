\begin{frame}[t,shrink=8]
\NotesTitle{Notes \& Steps}
\KeyIdea{Undo add/subtract first, then undo multiply/divide. Do the same to both sides.}
\vspace{-0.15em}
\footnotesize
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{\(2x+3=11\) (add then multiply)}{Step 1: Subtract 3
\[2x+3-3 = 11-3\]
\[2x = 8\]
Step 2: Divide by 2
\[\frac{2x}{2} = \frac{8}{2}\]
\[x = 4\]
Check: \(2\times4+3=11\)}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{\(5y-7=18\) (subtract then multiply)}{Step 1: Add 7
\[5y-7+7 = 18+7\]
\[5y = 25\]
Step 2: Divide by 5
\[\frac{5y}{5} = \frac{25}{5}\]
\[y = 5\]
Check: \(5\times5-7=18\)}
\end{column}
\end{columns}
\GreenDivider
\vspace{-0.1em}
\footnotesize
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{\(\frac{b}{2}+5=9\) (add then divide)}{Step 1: Subtract 5
\[\frac{b}{2}+5-5 = 9-5\]
\[\frac{b}{2} = 4\]
Step 2: Multiply by 2
\[\frac{b}{2}\times2 = 4\times2\]
\[b = 8\]
Check: \(8\div2+5=9\)}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{\(\frac{c}{3}-1=4\) (subtract then divide)}{Step 1: Add 1
\[\frac{c}{3}-1+1 = 4+1\]
\[\frac{c}{3} = 5\]
Step 2: Multiply by 3
\[\frac{c}{3}\times3 = 5\times3\]
\[c = 15\]
Check: \(15\div3-1=4\)}
\end{column}
\end{columns}
\GreenDivider
\vspace{-0.1em}
\footnotesize
\begin{columns}[b,onlytextwidth]
\begin{column}{0.40\textwidth}
\TryThese{\begin{enumerate}
  \item \(2x+3=11\)
  \item \(3a-4=14\)
  \item \(\frac{b}{2}+5=9\)
  \item \(\frac{c}{3}-1=4\)
\end{enumerate}}
\end{column}
\begin{column}{0.55\textwidth}
\CommonMistake{Undoing in wrong order. For \(2x+3=11\), subtract 3 first, \emph{then} divide by 2. Doing divide first gives \(x+3=5.5\) which is incorrect.}
\end{column}
\end{columns}
