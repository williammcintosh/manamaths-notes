\KeyIdea{Metric prefixes scale a unit by powers of 10. \textbf{kilo} means 1000 times bigger, \textbf{centi} means 100 times smaller, and \textbf{milli} means 1000 times smaller.}
\SectionTitle{Steps}
\StepLine{1}{Decide whether you are changing to a bigger or smaller unit.}
\StepLine{2}{Use powers of 10: metres \(\leftrightarrow\) centimetres \(\leftrightarrow\) millimetres.}
\StepLine{3}{Multiply when changing to a smaller unit.}
\StepLine{4}{Divide when changing to a bigger unit.}
\end{column}
\begin{column}{0.48\textwidth}
\SectionTitle{Key facts}
\begin{itemize}
  \item \(1\text{ km}=1000\text{ m}\)
  \item \(1\text{ m}=100\text{ cm}\)
  \item \(1\text{ cm}=10\text{ mm}\)
  \item \(1\text{ L}=1000\text{ mL}\)
  \item \(1\text{ kg}=1000\text{ g}\)
\end{itemize}
\end{column}
\end{columns}
