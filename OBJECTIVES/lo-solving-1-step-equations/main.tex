\begin{MMGreenBox}
\small\textbf{Key idea:} Do the inverse to both sides. 
+ $\leftrightarrow$ --, \quad $\times$ $\leftrightarrow$ $\div$
\end{MMGreenBox}
\vspace{-0.2em}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.23\textwidth}
\centering\footnotesize\textbf{Addition}\\
$x+5=12$\\
$x+5-5=12-5$\\
$x=7$\\
{\scriptsize Check: $7+5=12$}
\end{column}
\begin{column}{0.23\textwidth}
\centering\footnotesize\textbf{Subtraction}\\
$x-3=8$\\
$x-3+3=8+3$\\
$x=11$\\
{\scriptsize Check: $11-3=8$}
\end{column}
\begin{column}{0.23\textwidth}
\centering\footnotesize\textbf{Multiplication}\\
$3x=15$\\
$3x\div3=15\div3$\\
$x=5$\\
{\scriptsize Check: $3\times5=15$}
\end{column}
\begin{column}{0.23\textwidth}
\centering\footnotesize\textbf{Division}\\
$\frac{x}{4}=7$\\
$\frac{x}{4}\times4=7\times4$\\
$x=28$\\
{\scriptsize Check: $28\div4=7$}
\end{column}
\end{columns}
\vspace{-0.02em}
