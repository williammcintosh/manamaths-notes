\begin{frame}[t,shrink=8]
\NotesTitle{Topic 1 LO1 — Place Values and Decimals}
\footnotesize
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\KeyIdea{A digit's value depends on its place. The same digit can mean very different amounts in different positions.}
\SectionTitle{Steps}
\StepLine{1}{Look at the digit you care about.}
\StepLine{2}{Name its place: ones, tens, tenths, hundredths, and so on.}
\StepLine{3}{Write its value, not just the digit.}
\StepLine{4}{Compare left to right.}
\end{column}
\begin{column}{0.48\textwidth}
\SectionTitle{Place-value reminder}
\begin{itemize}
  \setlength{\itemsep}{0.12em}
  \item In \(347\), the \(4\) means \(40\).
  \item In \(5.3\), the \(3\) means \(0.3\).
  \item In \(2.74\), the \(7\) means \(0.7\).
  \item In \(2.74\), the \(4\) means \(0.04\).
\end{itemize}
\end{column}
\end{columns}
\GreenDivider
\vfill
\begin{columns}[b,onlytextwidth]
\begin{column}{0.48\textwidth}
\CommonMistake{Saying the value is just the digit. In \(862\), the digit is \(6\), but its value is \(60\). In \(2.74\), the digit is \(7\), but its value is \(0.7\).}
\end{column}
\begin{column}{0.48\textwidth}
\TryThese{\begin{enumerate}
  \item What is the value of the \(4\) in \(347\)?
  \item What is the value of the \(5\) in \(5.3\)?
  \item What is the value of the \(4\) in \(3.48\)?
\end{enumerate}}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[t,shrink=12]
\NotesTitle{Topic 1 LO1 — Place Values and Decimals}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\WorkedExample{Example 1: value of a digit}{Find the value of the \(6\) in \(862\).\[862 = 800 + 60 + 2\]So the value of the \(6\) is \(60\).}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 2: decimal place value}{Find the value of the \(7\) in \(2.74\).\[2.74 = 2 + 0.7 + 0.04\]So the value of the \(7\) is \(0.7\).}
\end{column}
\end{columns}
\GreenDivider
\TryThese{\begin{enumerate}
  \item Find the value of the \(9\) in \(291\).
  \item Find the value of the \(6\) in \(4.63\).
  \item Find the value of the \(2\) in \(8.205\).
\end{enumerate}}
\end{frame}

\begin{frame}[t,shrink=12]
\NotesTitle{Topic 1 LO1 — Place Values and Decimals}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\SectionTitle{Comparing numbers}
\small When two numbers have the same whole-number part, compare the decimal places next.\normalsize
\WorkedExample{Example 3: compare decimals}{Which is greater: \(0.8\) or \(0.08\)?\[0.8 = 8\text{ tenths}\]\[0.08 = 8\text{ hundredths}\]A tenth is bigger than a hundredth, so \(0.8 > 0.08\).}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 4: order numbers}{Order \(3.2,\ 2.9,\ 3.02\) from smallest to largest.\[2.9 < 3.02 < 3.2\]So the order is \(2.9,\ 3.02,\ 3.2\).}
\end{column}
\end{columns}
\GreenDivider
\CommonMistake{Thinking more digits means a bigger decimal. For example, \(0.08\) is not bigger than \(0.8\). It is smaller because 8 hundredths is less than 8 tenths.}
\end{frame}

\begin{frame}[t,shrink=12]
\NotesTitle{Topic 1 LO1 — Place Values and Decimals}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\SectionTitle{Partitioning numbers}
\small Split a number into place-value parts.\normalsize
\WorkedExample{Example 5: whole number}{Partition \(407\).\[407 = 400 + 0 + 7\]Usually we write this as \(407 = 400 + 7\).}
\end{column}
\begin{column}{0.48\textwidth}
\WorkedExample{Example 6: decimal number}{Partition \(3.48\).\[3.48 = 3 + 0.4 + 0.08\]That is 3 ones, 4 tenths, and 8 hundredths.}
\end{column}
\end{columns}
\GreenDivider
\TryThese{\begin{enumerate}
  \item Which is greater: \(45\) or \(54\)?
  \item Order from smallest to largest: \(0.5,\ 0.05,\ 0.15\).
  \item Partition \(4.7\).
\end{enumerate}}
